### Row Level Security Policies

```sql

ON photos FOR SELECT
TO authenticated
USING (
  user_id = auth.uid()
  AND is_deleted = FALSE

CREATE POLICY "admin_view_all"
ON photos FOR SELECT
TO authenticated
USING (
  EXISTS (
    SELECT 1 FROM profiles
    WHERE id = auth.uid()
    AND role IN ('admin', 'super_admin')
  )
```
